    /* ==============================================================
       0. WAIT FOR DOM TO BE FULLY LOADED
       ==============================================================
       We wrap everything in DOMContentLoaded so the HTML elements
       exist before we try to access them.
       ============================================================== */
    document.addEventListener('DOMContentLoaded', () => {

      /* ==============================================================
         1. UTILITY HELPERS
         ==============================================================
         Small reusable functions used throughout the code.
         ============================================================== */

      /**
       * Returns a random number between min and max.
       * @param {number} min - Lower bound (inclusive)
       * @param {number} max - Upper bound (exclusive)
       */
      function rand(min, max) {
        return Math.random() * (max - min) + min;
      }

      /**
       * Checks if the user prefers reduced motion.
       * We'll skip heavy animations if true.
       */
      const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

      /**
       * Detect if the device is likely mobile (for performance tuning).
       * We reduce particle counts on mobile to keep things smooth.
       */
      const isMobile = window.innerWidth < 768;

      /* ==============================================================
         2. BACKGROUND PARTICLES (CSS-animated floating dots)
         ==============================================================
         We create tiny <div> elements positioned randomly and let
         CSS @keyframes "floatUp" handle the animation.
         Fewer particles on mobile for performance.
         ============================================================== */
      (function createBgParticles() {
        if (prefersReducedMotion) return;      // skip if reduced motion

        const container = document.getElementById('bgParticles');
        const count = isMobile ? 15 : 35;      // fewer on mobile

        for (let i = 0; i < count; i++) {
          const p = document.createElement('div');
          p.className = 'bg-particle';
          /* Randomize position, size, speed, and delay */
          p.style.left            = rand(0, 100) + '%';
          p.style.bottom          = rand(-20, 10) + '%';
          p.style.width           = rand(1, 3) + 'px';
          p.style.height          = p.style.width;
          p.style.animationDuration = rand(12, 30) + 's';
          p.style.animationDelay    = rand(0, 15) + 's';
          /* Alternate colors for variety */
          if (Math.random() > 0.7) p.style.background = 'var(--neon-magenta)';
          if (Math.random() > 0.9) p.style.background = 'var(--transport-gold)';
          container.appendChild(p);
        }
      })();

      /* ==============================================================
         3. PORTAL CANVAS ANIMATION
         ==============================================================
         Draws a ring of particles that swirl and converge, creating
         the "dimensional portal" effect behind the DOC911 title.

         HOW IT WORKS:
         - We define a Particle class.  Each particle orbits the
           center in a circle, with slight oscillation in radius.
         - On every frame we clear the canvas, update each particle,
           and draw it as a tiny glowing circle.
         - requestAnimationFrame gives us smooth 60fps rendering.
         ============================================================== */
      (function initPortalCanvas() {
        const canvas = document.getElementById('portalCanvas');
        const ctx    = canvas.getContext('2d');

        /* --- Resize canvas to fill its container --- */
        function resize() {
          canvas.width  = canvas.offsetWidth;
          canvas.height = canvas.offsetHeight;
        }
        resize();
        window.addEventListener('resize', resize);

        /* Particle count – fewer on mobile */
        const PARTICLE_COUNT = isMobile ? 80 : 200;

        /* Center of the canvas (portal center) */
        let cx, cy;
        function updateCenter() {
          cx = canvas.width / 2;
          cy = canvas.height / 2;
        }
        updateCenter();
        window.addEventListener('resize', updateCenter);

        /* --- Particle class --- */
        class Particle {
          constructor() {
            this.reset();
          }

          reset() {
            /* Each particle orbits at a random radius and angle */
            this.angle     = rand(0, Math.PI * 2);   // current angle
            this.radius    = rand(60, Math.min(cx, cy) * 0.7); // orbit radius
            this.baseRadius = this.radius;
            this.speed     = rand(0.003, 0.015);      // angular speed
            this.size      = rand(0.5, 2.5);          // dot size
            this.pulse     = rand(0, Math.PI * 2);    // phase offset for pulsing
            this.pulseSpeed = rand(0.02, 0.06);
            /* Color: mostly cyan, some magenta, some gold */
            const r = Math.random();
            if (r > 0.85)      this.color = '255,0,255';     // magenta
            else if (r > 0.75) this.color = '255,215,0';     // gold
            else               this.color = '0,240,255';     // cyan
          }

          update() {
            /* Rotate around center */
            this.angle += this.speed;

            /* Oscillate the radius to create a breathing effect */
            this.pulse += this.pulseSpeed;
            this.radius = this.baseRadius + Math.sin(this.pulse) * 15;

            /* Convert polar coords to cartesian */
            this.x = cx + Math.cos(this.angle) * this.radius;
            this.y = cy + Math.sin(this.angle) * this.radius;
          }

          draw(ctx) {
            const alpha = 0.3 + Math.sin(this.pulse) * 0.3;  // pulsing opacity
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(${this.color}, ${alpha})`;
            ctx.fill();

            /* Small glow around each particle */
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size * 3, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(${this.color}, ${alpha * 0.1})`;
            ctx.fill();
          }
        }

        /* Create particle array */
        const particles = [];
        for (let i = 0; i < PARTICLE_COUNT; i++) {
          particles.push(new Particle());
        }

        /* --- Draw the portal ring (subtle outer glow ring) --- */
        function drawPortalRing(time) {
          const ringRadius = Math.min(cx, cy) * 0.45;
          /* Outer glow */
          ctx.beginPath();
          ctx.arc(cx, cy, ringRadius, 0, Math.PI * 2);
          ctx.strokeStyle = `rgba(0,240,255, ${0.08 + Math.sin(time * 0.001) * 0.04})`;
          ctx.lineWidth = 2;
          ctx.stroke();
          /* Inner ring */
          ctx.beginPath();
          ctx.arc(cx, cy, ringRadius * 0.6, 0, Math.PI * 2);
          ctx.strokeStyle = 'rgba(255,0,255, 0.04)';
          ctx.lineWidth = 1;
          ctx.stroke();
        }

        /* --- Main animation loop --- */
        let animId;
        function animate(time) {
          /* Clear entire canvas each frame */
          ctx.clearRect(0, 0, canvas.width, canvas.height);

          /* Draw central gradient glow */
          const grad = ctx.createRadialGradient(cx, cy, 0, cx, cy, Math.min(cx, cy) * 0.5);
          grad.addColorStop(0, 'rgba(0,240,255,0.04)');
          grad.addColorStop(0.5, 'rgba(255,0,255,0.02)');
          grad.addColorStop(1, 'transparent');
          ctx.fillStyle = grad;
          ctx.fillRect(0, 0, canvas.width, canvas.height);

          /* Draw the ring guides */
          drawPortalRing(time);

          /* Update & draw every particle */
          for (const p of particles) {
            p.update();
            p.draw(ctx);
          }

          /* Draw connecting lines between nearby particles (creates web effect) */
          if (!isMobile) {
            for (let i = 0; i < particles.length; i++) {
              for (let j = i + 1; j < particles.length; j++) {
                const dx = particles[i].x - particles[j].x;
                const dy = particles[i].y - particles[j].y;
                const dist = Math.sqrt(dx * dx + dy * dy);
                if (dist < 80) {
                  ctx.beginPath();
                  ctx.moveTo(particles[i].x, particles[i].y);
                  ctx.lineTo(particles[j].x, particles[j].y);
                  ctx.strokeStyle = `rgba(0,240,255, ${0.08 * (1 - dist / 80)})`;
                  ctx.lineWidth = 0.5;
                  ctx.stroke();
                }
              }
            }
          }

          animId = requestAnimationFrame(animate);
        }

        /* Don't start the heavy canvas animation if reduced motion */
        if (!prefersReducedMotion) {
          animate(0);
        }
      })();

      /* ==============================================================
         4. BOOT SEQUENCE
         ==============================================================
         Simulates a terminal boot-up.  Each line appears with a delay,
         giving the illusion of a system initializing.
         ============================================================== */
      (function runBootSequence() {

        /* Array of boot messages.  Each has text and delay (ms) */
        const messages = [

        messages.forEach(({ text, delay }) => {
          setTimeout(() => {
            const line = document.createElement('div');
            line.className = 'boot-line';
            /* Add blinking cursor at the end */
            line.innerHTML = text + ' <span class="cursor">█</span>';
            container.appendChild(line);

            const allCursors = container.querySelectorAll('.cursor');
            if (allCursors.length > 1) {
              allCursors[allCursors.length - 2].remove();

        });
      })();

      /* ==============================================================
         5. SCROLL-TRIGGERED ANIMATIONS (Intersection Observer)
         ==============================================================
         The Intersection Observer API watches elements as they scroll
         into view.  When an element enters the viewport, we add a
         "visible" class to trigger its CSS entrance animation.

         WHY Intersection Observer?
           - Much more performant than listening to scroll events.
           - No need for debouncing or throttling.
           - Built into modern browsers.
         ============================================================== */
      (function initScrollAnimations() {
        /* Select all elements that should animate on scroll */
        const targets = document.querySelectorAll('.panel, .story-block, .timeline-item, .signal-beacon');

        /* Configuration:
           - threshold: 0.15 means 15% of the element must be visible
           - rootMargin: slight offset so animation starts just before visible */
        const observer = new IntersectionObserver((entries) => {
          entries.forEach(entry => {
            if (entry.isIntersecting) {
              entry.target.classList.add('visible');
              /* Once animated in, stop observing (one-time animation) */
              observer.unobserve(entry.target);
            }
          });
        }, {
          threshold: 0.15,
          rootMargin: '0px 0px -50px 0px'
        });

        /* Observe each target element */
        targets.forEach(el => observer.observe(el));
      })();

      /* ==============================================================
         6. COORDINATE COUNTER ANIMATION
         ==============================================================
         The dimensional coordinates in the dashboard update
         periodically to random values, giving a "live data" feel.

      (function animateCoordinates() {

        function updateCoords() {
          coordEls.forEach(el => {
            /* Generate a random coordinate value based on axis */
            const axis = el.dataset.coord;
            let value;
            if (axis === 't') {
              /* Temporal index: slowly incrementing */
              value = (parseFloat(el.textContent) + rand(0.001, 0.01)).toFixed(3);
            } else {
              /* Spatial: random fluctuation */
              value = rand(0, 10).toFixed(3);
            }
            el.textContent = value;
          });
        }

        /* Update every 2 seconds */
        setInterval(updateCoords, 2000);
      })();

      (function initCommsFeed() {
        const feed = document.getElementById('commsFeed');

        /* Pool of possible messages */
        const msgPool = [
          { text: 'Dimensional rift stable. No anomalies.', type: '' },
          { text: 'Signal received from Dimension Cyan-7.', type: '' },

          { text: '⚠ Shield fluctuation — compensating.', type: 'alert-msg' },
          { text: 'Incoming hail from Dimension Magenta-∞.', type: 'gold-msg' },
          { text: 'Navigation array recalibrated.', type: '' },
          { text: 'Temporal index synchronized.', type: '' },
          { text: '⚠ Unknown energy signature detected!', type: 'alert-msg' },

        function addMessage() {

      /* ==============================================================
         8. NAVIGATION — Scroll Spy & Hamburger Menu
         ==============================================================
         • Scroll Spy: highlights the nav link for whichever section
           is currently in view.
         • Hamburger: toggles the mobile menu open/closed.
         ============================================================== */
      (function initNavigation() {
        const hamburger = document.getElementById('hamburger');
        const navLinks  = document.getElementById('navLinks');
        const links     = navLinks.querySelectorAll('a:not(.nav-transport-btn)');
        const sections  = document.querySelectorAll('section[id]');

        /* --- Hamburger toggle --- */
        hamburger.addEventListener('click', () => {
          hamburger.classList.toggle('open');
          navLinks.classList.toggle('open');
          const expanded = hamburger.classList.contains('open');
          hamburger.setAttribute('aria-expanded', expanded);
        });

        /* Close menu when a link is clicked (mobile) */
        navLinks.querySelectorAll('a').forEach(link => {
          link.addEventListener('click', () => {
            hamburger.classList.remove('open');
            navLinks.classList.remove('open');
            hamburger.setAttribute('aria-expanded', 'false');
          });
        });

        /* --- Scroll Spy with Intersection Observer --- */
        const spyObserver = new IntersectionObserver((entries) => {
          entries.forEach(entry => {
            if (entry.isIntersecting) {
              /* Remove active from all, add to matching link */
              links.forEach(l => l.classList.remove('active'));
              const activeLink = navLinks.querySelector(`a[href="#${entry.target.id}"]`);
              if (activeLink) activeLink.classList.add('active');
            }
          });
        }, {
          /* 
           * rootMargin: top offset pulls trigger up into the nav bar area,
           * bottom offset means only the top portion of the section needs
           * to be visible.
           */
          rootMargin: '-30% 0px -60% 0px'
        });

        sections.forEach(sec => spyObserver.observe(sec));
      })();

      /* ==============================================================
         9. TRANSPORTER ACTIVATION SEQUENCE
         ==============================================================
         The flagship interaction!  When the user clicks "Initiate
         Transport", we run through 5 phases:

           Phase 1: CHARGE    — Particle buildup in overlay canvas
           Phase 2: LOCK      — Coordinate lock + rings tighten
           Phase 3: TRANSPORT  — Chromatic aberration, intense glow
           Phase 4: ARRIVAL   — Particle burst outward
           Phase 5: COMPLETE  — Calm fade, success message

         Each phase lasts ~2 seconds.  The overlay covers the full
         screen and uses a <canvas> for particle effects.
         ============================================================== */
      (function initTransporter() {
        const overlay       = document.getElementById('transporterOverlay');
        const phaseText     = document.getElementById('transportPhaseText');
        const subText       = document.getElementById('transportSubText');
        const activateBtn   = document.getElementById('transportActivateBtn');
        const navBtn        = document.getElementById('navTransportBtn');
        const closeBtn      = document.getElementById('transportClose');
        const canvas        = document.getElementById('transportCanvas');
        const ctx           = canvas.getContext('2d');
        const phaseDots     = document.querySelectorAll('.phase-dot .circle');
        const energyRings   = document.querySelectorAll('.energy-ring');

        let isTransporting = false;   // prevents double-activation
        let transportAnimId;          // animation frame ID for cleanup

        /* --- Resize transport canvas --- */
        function resizeTransport() {
          canvas.width  = window.innerWidth;
          canvas.height = window.innerHeight;
        }
        resizeTransport();
        window.addEventListener('resize', resizeTransport);

        /* --- Transport Particle class --- */
        class TransportParticle {
          constructor(cx, cy, phase) {
            this.cx = cx;
            this.cy = cy;
            this.phase = phase;
            this.reset();
          }

          reset() {
            this.angle = rand(0, Math.PI * 2);
            /* In early phases, particles converge inward.
               In phase 4 (arrival), they burst outward. */
            if (this.phase >= 4) {
              this.radius = rand(0, 20);
              this.speed  = rand(2, 8);          // expand outward
            } else {
              this.radius = rand(100, Math.max(this.cx, this.cy));
              this.speed  = rand(-3, -0.5);      // converge inward
            }
            this.size = rand(1, 3);
            this.alpha = rand(0.3, 1);
            this.alive = true;

            const r = Math.random();
            if (r > 0.7) this.color = '255,0,255';
            else if (r > 0.4) this.color = '255,215,0';
            else this.color = '0,240,255';
          }

          update() {
            this.radius += this.speed;
            this.angle  += 0.01;
            this.x = this.cx + Math.cos(this.angle) * this.radius;
            this.y = this.cy + Math.sin(this.angle) * this.radius;

            /* Kill when too far out or too close to center */
            if (this.radius < 0 || this.radius > Math.max(this.cx, this.cy) * 1.5) {
              this.alive = false;
            }
          }

          draw(ctx) {
            if (!this.alive) return;
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(${this.color}, ${this.alpha})`;
            ctx.fill();
            /* Glow */
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size * 4, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(${this.color}, ${this.alpha * 0.15})`;
            ctx.fill();
          }
        }

        /* --- Phase definitions --- */
        const phases = [
          { name: 'CHARGING',     sub: 'Building quantum field energy...',        duration: 2000 },
          { name: 'LOCK',         sub: 'Dimensional coordinates locked.',         duration: 2000 },
          { name: 'TRANSPORTING', sub: 'Reality membrane breached — hold on!',    duration: 2500 },
          { name: 'ARRIVAL',      sub: 'Materializing in target dimension...',    duration: 2000 },
          { name: 'COMPLETE',     sub: 'Transport successful. Welcome, traveller.', duration: 2000 },
        ];

        /* --- Run the full sequence --- */
        function startTransport() {
          if (isTransporting) return;
          isTransporting = true;
          overlay.classList.add('active');
          resizeTransport();

          const cxT = canvas.width / 2;
          const cyT = canvas.height / 2;

          let currentPhase = 0;
          let particles = [];

          /* --- Advance to next phase --- */
          function nextPhase() {
            if (currentPhase >= phases.length) {
              /* Sequence complete — auto-close after a brief pause */
              setTimeout(closeTransport, 1500);
              return;
            }

            const phase = phases[currentPhase];
            phaseText.textContent = phase.name;
            subText.textContent   = phase.sub;

            /* Light up phase dot */
            phaseDots.forEach((dot, i) => {
              dot.classList.remove('active', 'done');
              if (i < currentPhase)  dot.classList.add('done');
              if (i === currentPhase) dot.classList.add('active');
            });

            /* Animate energy rings during charge & transport */
            if (currentPhase === 0 || currentPhase === 2) {
              energyRings.forEach(ring => {
                ring.style.animation = 'none';
                void ring.offsetWidth;   // force reflow
                ring.style.animation = 'ringExpand 2s ease-out forwards';
              });
            }

            /* Apply visual effects to overlay based on phase */
            if (currentPhase === 2) {
              /* Transport phase: intense chromatic aberration simulation */
              overlay.style.background = 'rgb(10,0,30)';
              overlay.style.boxShadow = 'inset 0 0 200px rgba(255,0,255,0.3), inset 0 0 100px rgba(0,240,255,0.15)';
            } else if (currentPhase === 3) {
              /* Arrival: bright flash then calm */
              overlay.style.background = 'rgb(200,200,255)';
              setTimeout(() => { overlay.style.background = 'rgb(0,0,0)'; }, 300);
            } else if (currentPhase === 4) {
              overlay.style.background = 'rgb(0,0,0)';
              overlay.style.boxShadow = 'none';
            }

            /* Spawn phase-specific particles */
            const count = isMobile ? 30 : 80;
            for (let i = 0; i < count; i++) {
              particles.push(new TransportParticle(cxT, cyT, currentPhase));
            }

            /* 🔊 AUDIO TRIGGER POINT: Play phase-specific sound
               switch(currentPhase) {
                 case 0: chargeSound.play(); break;
                 case 1: lockSound.play(); break;
                 case 2: transportSound.play(); break;
                 case 3: arrivalSound.play(); break;
               }
            */

            currentPhase++;
            setTimeout(nextPhase, phase.duration);
          }

          /* --- Particle animation loop --- */
          function animateTransport() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            /* Draw center glow */
            const grad = ctx.createRadialGradient(cxT, cyT, 0, cxT, cyT, 200);
            grad.addColorStop(0, 'rgba(255,215,0,0.06)');
            grad.addColorStop(0.5, 'rgba(0,240,255,0.03)');
            grad.addColorStop(1, 'transparent');
            ctx.fillStyle = grad;
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            /* Update & draw particles */
            particles = particles.filter(p => p.alive);
            for (const p of particles) {
              p.update();
              p.draw(ctx);
            }

            transportAnimId = requestAnimationFrame(animateTransport);
          }

          animateTransport();
          nextPhase();
        }

        /* --- Close / reset the transporter --- */
        function closeTransport() {
          overlay.classList.remove('active');
          overlay.style.background = '';
          overlay.style.boxShadow = '';
          isTransporting = false;
          cancelAnimationFrame(transportAnimId);
          ctx.clearRect(0, 0, canvas.width, canvas.height);

          /* Reset phase dots */
          phaseDots.forEach(dot => dot.classList.remove('active', 'done'));

          /* Reset text */
          phaseText.textContent = 'CHARGING';
          subText.textContent   = 'Initializing quantum field...';
        }

        /* --- Event listeners --- */
        activateBtn.addEventListener('click', startTransport);
        navBtn.addEventListener('click', (e) => {
          e.preventDefault();
          startTransport();
        });
        closeBtn.addEventListener('click', closeTransport);

        /* Close on Escape key (accessibility) */
        document.addEventListener('keydown', (e) => {
          if (e.key === 'Escape' && isTransporting) closeTransport();
        });
      })();

      /* ==============================================================
         10. PARALLAX EFFECT ON ABOUT SECTION
         ==============================================================
         Subtle parallax: the beacon circle moves slower than scroll,
         creating a depth illusion.  Uses a debounced scroll listener.
         ============================================================== */
      (function initParallax() {
        if (prefersReducedMotion || isMobile) return;   // skip on mobile

        const beacon = document.querySelector('.about-beacon');
        if (!beacon) return;

        let ticking = false;

        window.addEventListener('scroll', () => {
          if (!ticking) {
            requestAnimationFrame(() => {
              const scrollY = window.scrollY;
              /* Move beacon at 30% of scroll speed for depth effect */
              beacon.style.transform = `translateY(${scrollY * 0.15}px)`;
              ticking = false;
            });
            ticking = true;          // throttle to 1 update per frame
          }
        });
      })();

      })();

      /* ==============================================================
         12. RADAR BLIP MOVEMENT
         ==============================================================
         Periodically repositions radar blips to random locations
         within the radar circle, simulating detected objects moving.
         ============================================================== */
      (function animateRadarBlips() {
        const blips = document.querySelectorAll('.radar-blip');

        function moveBlips() {
          blips.forEach(blip => {
